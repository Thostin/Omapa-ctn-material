% --- List of problem keys (in order of appearance) ---
\def\problemlist{}

% --- Command to be used INSIDE the problem environment ---
\newcommand{\solutionlabel}[1]{%
  \xdef\currentproblemkey{#1}%               remember key for the hook
  \xdef\currentproblemnumber{\theproblem}%   store problem number
  \expandafter\xdef\csname problemnumber@#1\endcsname{\currentproblemnumber}%
  \ifinlist{#1}{\problemlist}{}{\listadd{\problemlist}{#1}}% add key to list
}

% --- Hook: append "Sol: p.yy" after every problem ---
\AfterEndEnvironment{problem}{%
  \ifdefined\currentproblemkey
    \unskip\space Sol: \pageref{solpage:\currentproblemkey}%
    \global\let\currentproblemkey\undefined
    \global\let\currentproblemnumber\undefined
  \fi
}

% --- Redefine the solution environment to store the solution and place a label ---
\NewEnviron{solution}[1]{%
  \global\expandafter\let\csname solcontent@#1\endcsname\BODY   % store body
  \label{solpage:#1}%                                           % page reference
}

% --- Print all solutions in problem order ---
\newcommand{\printsolutions}{%
  \forlistloop{\printsolution}{\problemlist}
}
\newcommand{\printsolution}[1]{%
  \par\vspace{\baselineskip}%
  \noindent\textbf{Solution to problem \csname problemnumber@#1\endcsname.}\par
  \csname solcontent@#1\endcsname\par
}
