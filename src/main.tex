% Usage: \begin{solution}{<key>} ... \end{solution}
\NewDocumentEnvironment{solutiontoproblem}{m +b}
  {
    % store key in order list
    \seq_gput_right:Nn \g_solution_order_seq {#1}
    % store body under that key
    \prop_gput:Nnn \g_solution_body_prop {#1} {#2}
  }
  {}

% Print all stored solutions, in the order they were added
\NewDocumentCommand{\printsolutions}{}%
  {
    \seq_map_inline:Nn \g_solution_order_seq
      {
        \par\bigskip
        \section*{Solución~al~problema~\ref{##1}}
        \label{solution:##1}
        \prop_item:Nn \g_solution_body_prop {##1}
      }
  }
