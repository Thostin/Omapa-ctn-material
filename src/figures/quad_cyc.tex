\begin{asy}
	pair A = dir(80);
	pair B = dir(160);
	pair C = dir(210);
	pair D = dir(330);

	pair X = 1.3 * (B - C) + C;
	pair Y = 1.2 * (A - D) + D;
	draw(X--B);
	draw(A--Y);

	draw(A--B--C--D--cycle);

	draw(A--C);
	draw(B--D);

	draw(unitcircle);

	dot("$A$", A, NE);
	dot("$B$", B, NW);
	dot("$C$", C, SW);
	dot("$D$", D, SE);

	markangle(A, B, X, radius=10, blue);
	markangle(A, D, C, radius=10, blue);
	markangle(D, C, B, radius=10, red);
	markangle(Y, A, B, radius=10, red);

	markangle(C, B, D, radius=13, green);
	markangle(C, A, D, radius=13, green);
\end{asy}
